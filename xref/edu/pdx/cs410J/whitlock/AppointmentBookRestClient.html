1   package edu.pdx.cs410J.whitlock;
2   
3   import com.google.common.annotations.VisibleForTesting;
4   import edu.pdx.cs410J.web.HttpRequestHelper;
5   
6   import java.io.IOException;
7   
8   /**
9    * A helper class for accessing the rest client
10   */
11  public class AppointmentBookRestClient extends HttpRequestHelper
12  {
13      private static final String WEB_APP = "apptbook";
14      private static final String SERVLET = "appointments";
15  
16      private final String url;
17  
18  
19      /**
20       * Creates a client to the appointment book REST service running on the given host and port
21       * @param hostName The name of the host
22       * @param port The port
23       */
24      public AppointmentBookRestClient( String hostName, int port )
25      {
26          this.url = String.format( "http://%s:%d/%s/%s", hostName, port, WEB_APP, SERVLET );
27      }
28  
29      /**
30       * Returns all keys and values from the server
31       */
32      public Response getAllKeysAndValues() throws IOException
33      {
34          return get(this.url );
35      }
36  
37      /**
38       * Returns all values for the given key
39       */
40      public Response getValues( String key ) throws IOException
41      {
42          return get(this.url, "key", key);
43      }
44  
45      public Response addKeyValuePair( String key, String value ) throws IOException
46      {
47          return postToMyURL("key", key, "value", value);
48      }
49  
50      @VisibleForTesting
51      Response postToMyURL(String... keysAndValues) throws IOException {
52          return post(this.url, keysAndValues);
53      }
54  
55      public Response removeAllMappings() throws IOException {
56          return delete(this.url);
57      }
58  
59      public Response prettyPrintAppointmentBook(String owner) throws IOException {
60          return get(this.url, "owner", owner);
61      }
62  
63      public Response createAppointment(String owner, String description, String beginTime, String endTime) throws IOException {
64          return post(this.url, "owner", owner, "description", description, "beginTime", beginTime, "endTime", endTime);
65      }
66  }
